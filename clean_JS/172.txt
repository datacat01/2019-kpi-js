function trackModalTabs() {
	modal = document.querySelector('#SaksBagOverlay');
	var focusableElements = modal.querySelectorAll(focusableElementsSelector);
	firstTabStop = focusableElements[0];
	lastTabStop = focusableElements[focusableElements.length - 1];
	modal.addEventListener('keydown', keyDownEventListenerCallback);
	firstTabStop.focus();
}

function keyDownEventListenerCallback(e) {
  if (e.keyCode === TAB) {
    if (e.shiftKey && document.activeElement === firstTabStop) {
      e.preventDefault();
      lastTabStop.focus();
    } else if (!e.shiftKey && document.activeElement === lastTabStop) {
      e.preventDefault();
      firstTabStop.focus();
    }
  } else if (e.keyCode === ESC) {
    SaksBagClose();
  }
}
