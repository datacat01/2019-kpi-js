    function AnalyticsTrackEvent(category, action, opt_label, opt_value) {
        // section: Header_Analytics_ascx.AnalyticsTrackEvent
        try
        {
            // before we do anything, check if this is a search event with no opt_label, if so, return
            if( category === 'Search' && opt_label === '' )
            {
                return;
            }
            // push the event to the JS cookie set, it will get sent to Google with the next page load
            return _pushEventToGoogle(category, action, opt_label, opt_value);
        }
        catch(err){
            _logException(err, 'Header_Analytics_ascx.AnalyticsTrackEvent');
        }
    }


    /* push event to google */
    function _pushEventToGoogle(category, action, opt_label, opt_value, opt_noninteraction)
    {

        // first check that both are set and silently fail if not
        if (typeof category == "undefined" || category == null || category.length == 0) {
            return true;
        }
        if (typeof action == "undefined" || action == null || action.length == 0) {
            return true;
        }
        if (typeof opt_label == "undefined" || opt_label == null || opt_label.length == 0) {
            opt_label = '';
        }
        if (typeof opt_value == "undefined" || opt_value == null || opt_value.length == 0 || typeof(opt_value)!='number') {
            opt_value = -1;
        }

        if (typeof opt_noninteraction == "undefined" || opt_noninteraction == null || opt_noninteraction.length == 0 || typeof (opt_noninteraction) != 'boolean') {
            opt_noninteraction = true;
        }



        /* below, use the correct call to _trackEvent based on what we have as input */
        if (opt_value > -1) {
            // section: Header_Analytics_ascx._pushEventToGoogle_ALL
            try {
                dataLayer.push({ 'event': 'trackEvent', 'eventCategory': category, 'eventAction': action, 'eventLabel': opt_label, 'eventValue': opt_value });
            }
            catch (err) {
                _logException(err, 'Header_Analytics_ascx._pushEventToGoogle_ALL');
            }

            return true;
          }

        if (opt_label != '') {
            // section: Header_Analytics_ascx._pushEventToGoogle_OPTLABEL
            try {
                dataLayer.push({ 'event': 'trackEvent', 'eventCategory': category, 'eventAction': action, 'eventLabel': opt_label, 'eventValue': '' });
            }
            catch (err) {
                _logException(err, 'Header_Analytics_ascx._pushEventToGoogle_OPTLABEL');
            }

            return true;
        }

        // section: Header_Analytics_ascx._pushEventToGoogle_CAT_ACTION
        try {
            dataLayer.push({ 'event': 'trackEvent', 'eventCategory': category, 'eventAction': action, 'eventLabel': '', 'eventValue': '' });
        }
        catch (err) {
            _logException(err, 'Header_Analytics_ascx._pushEventToGoogle_CAT_ACTION');
        }

        return true;
    }

    function _logPushTo(eventArr, pushTo)
    {
        var errCacheBuster = parseInt(Math.random()*99999999);  // cache buster
        var category = eventArr[0];
        var action = eventArr[1];
        var opt_label = eventArr[2];
        var opt_value = eventArr[3];
        var qs = 'pushTo=' + escape(pushTo)
                + '&category=' + escape(category)
                + '&action=' + escape(action)
                + '&opt_label=' + escape(opt_label)
                + '&opt_value=' + escape(opt_value)
                + '&cacheBuster=' + errCacheBuster;
        var url = '/assets/images/transparent.gif?' + qs;

        /* if we can append this now, do so, otherwise when body is loaded (because of IE problems, only do this if jQuery is loaded) */
        if (typeof jQuery != 'undefined') {
            if( $('body') != null ){
                $('body').append('<img src=\'' + url + '\' />');
            }
            else{
                $(document).ready(function() {
                    $('body').append('<img src=\'' + url + '\' />');
                });
            }
        }


    }


    /* _logException - causes image request to be emitted that records this
                       exception in the IIS logs as an image request for transparent.gif */
    function _logException(err, location){

        var errCacheBuster = parseInt(Math.random()*99999999);  // cache buster
        var errorMsg = 'no description on exception';
        if( err != null ){
            errorMsg = err.description;
        }
        if( errorMsg != null ){
            if( errorMsg.length > 1024 ){
                errorMsg = errorMsg.substring(0,1024);
            }
        }
        else{
            errorMsg = 'no description on exception';
        }

        var qs = 'err=' + escape(errorMsg)
                + '&loc=' + escape(location)
                + '&cacheBuster=' + errCacheBuster;
        var url = '/assets/images/transparent.gif?' + qs;

        /* this doesn't seem to work unless it is document.write instead of append */
        document.write('<img id="err" src=\'' + url + '\' />');
    }

	function Survey5_Init() {
		var pathname = window.location.pathname.toLowerCase();

		//ensure we are not interrupting a purchase
		if (!pathname.match(/.*(purchase|shipping|shoppingcart).*/)) {
			var survey5state = EGPGetCookie('survey5');

			if (survey5state == null || survey5state == 0) {
				EGPSetCookie('survey5', 1, 90, '/');
			}
			else if (survey5state == 1) {
				EGPSetCookie('survey5', 2, 90, '/');
				Survey5_Show();
			}
		}
	}
