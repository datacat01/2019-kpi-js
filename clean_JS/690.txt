
  function checkJqBootstrap() {
    if (window.jQuery && jQuery.ui) {
      if (/(faq)|(about)|(careers)|(drums-alive)|(drum-circle)|(in-our-stores)|(learn-and-discover)|(news-release)|(partners-and-awards)|(privacy-policy)|(promo)|(resources)|(service)|(shop-by-brand)/.test(window.location.href)||window.location.href.split("?")[0]=="https://www.westmusic.com/"||window.location.href.split("?")[0]=="https://www.westmusic.com/home"){
        if (!(/(music-books-resources)/.test(window.location.href))){
          var bootstrapjs = document.createElement('script');
          bootstrapjs.type = 'text/javascript';
          bootstrapjs.src = '//www.westmusic.com/content/css/Custom/bootstrap/js/bootstrap.min.js';
          bootstrapjs.defer = true;
          document.head.appendChild(bootstrapjs);
          var bootstrapcss = document.createElement('link');
          bootstrapcss.rel = 'stylesheet';
          bootstrapcss.href = '//www.westmusic.com/content/css/Custom/bootstrap/css/bootstrap.min.css';
          document.head.appendChild(bootstrapcss);
        }
      }
    } else {
      window.setTimeout(checkJqBootstrap, 100);
    }
  }
  checkJqBootstrap();
