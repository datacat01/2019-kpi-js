
    var g_tag_params = {
        ecomm_pagetype: 'other'

    
    
    };


    window.addEventListener('load',function(){
        var ecomm_prodid= '';
        var ecomm_pagetype= '';
        var ecomm_totalvalue= 0;
        var pathnameVal=window.location.pathname;
        var hrefVal=window.location.href;

        if(pathnameVal=='/') /* for home page */
        {
            ecomm_pagetype= 'home';
        }
        else if(jQuery('.product-details__title').length > 0)  /*  product page */
        {

            ecomm_prodid= window.g_tag_params.ecomm_prodid;
            ecomm_pagetype= 'product';
            ecomm_totalvalue= jQuery('[itemprop="price"]').attr('content');
        }
        else if(hrefVal.indexOf('/category/') != -1) /* for category page */
        {
            ecomm_pagetype= 'category';

        }
        else if(hrefVal.indexOf('/shopping-cart') != -1)  /*  cart page */
        {
            ecomm_pagetype= 'cart';
            ecomm_prodid = window.g_tag_params.ecomm_prodid;
            ecomm_totalvalue = window.g_tag_params.ecomm_totalvalue;

        }
        else if(hrefVal.indexOf('/checkout/checkout_success.php') != -1)
        {
            ecomm_pagetype= 'purchase';
        }
        else
        {
            ecomm_pagetype= 'other';
        }

        var google_tag_params = {
            ecomm_prodid:  ecomm_prodid,
            ecomm_pagetype: ecomm_pagetype,
            ecomm_totalvalue: parseFloat(ecomm_totalvalue)
        };

        window.google_trackConversion({
            google_conversion_id: 975259728,
            google_custom_params: {
                'ecomm_prodid': google_tag_params.ecomm_prodid,
                'ecomm_pagetype': google_tag_params.ecomm_pagetype,
                'ecomm_totalvalue':google_tag_params.ecomm_totalvalue
            },
            google_remarketing_only: true
        });
    });
