//handle the mobile menu opening and closing and also the double click action for the top level
//first click opens up the secondary nav
//second click goes to the actual page
jQuery('.firstLevel a').click(function(){
    var subMenuStyle = jQuery(this).parent().find(".sub-menu");

    if(subMenuStyle.length == 0)
    {
       //no submenu, so allow click
       return true;
    } 

    //set other submenus to have the plus icon
    loadElements();

    if(subMenuStyle.css("display") == "none")
    {
        var replaceString = jQuery(this).html().replace('fa fa-plus', 'fa fa-minus');
        jQuery(this).html(replaceString);
        jQuery(".sub-menu").slideUp();
        jQuery(this).parent().children("ul").toggle();
        return false;
    }
    else if(subMenuStyle.css("display") == "block")
    {
        //alert('should be changing to plus sign now');
        var replaceString = jQuery(this).html().replace('fa fa-minus', 'fa fa-plus');
        jQuery(this).html(replaceString);
        return true;
    }
    else
    {
        return true;
    }
});
});

//loads elements for the first time and reloads when other item is clicked.
function loadElements(){
jQuery('.firstLevel a').each(function(){
   var subMenuStyle = jQuery(this).parent().find(".sub-menu");
   if(subMenuStyle.length == 0)
   {
     //remove the items with no children
      var replaceString = jQuery(this).html().replace('fa fa-plus', '');
      jQuery(this).html(replaceString);
   } 
   else{
      var replaceString = jQuery(this).html().replace('fa fa-minus', 'fa fa-plus');
      jQuery(this).html(replaceString);
   }
});
}
