function pdSupportIsOpen() {
	try {
		var csDateTimeString = new Date().toLocaleString("en-US", {timeZone: "America/New_York"});
		var csDateTimestamp = Date.parse(csDateTimeString);
	} catch(e) {	//*** BAIL OUT FOR IE
		return false;
	}
	var csDateTime = new Date(csDateTimestamp);
	var csDay = csDateTime.getDay();		//*** 0 = Sunday, 6 = Saturday
	var csHour = csDateTime.getHours();
	var supportIsOpen = false;
	if(csDay == 0) {	// IF DAY OF WEEK IS SUNDAY, SUPPORT IS CLOSED
		supportIsOpen = false;
	} else if(csDay == 6) {	// IF DAY IS SATURDAY, SUPPORT IS OPEN 9am-6pm ET
		if(csHour >= 9 && csHour < 18){	//*** HOUR MUST BE LESS THAN 6PM (< 18) TO REPRESENT BEFORE 6PM
			supportIsOpen = true;
		}
	} else {	// IF DAY IS MONDAY - FRIDAY, SUPPORT IS OPEN 8am-6pm ET
		if(csHour >= 8 && csHour < 19) {
			supportIsOpen = true;
		}
	}
	return supportIsOpen;
}
function pdSupportIsOpenDisplay(jsTimestamp) {
	var pdIsOpen  = false;
	if(jsTimestamp == false) {
		pdIsOpen = false;
	} else if(jsTimestamp == true) {
		pdIsOpen = true;
	} else {
		pdIsOpen = pdSupportIsOpen();
	}

	if(pdIsOpen){
		jQuery(".headercontactphoneopen").show();
		jQuery(".headercontactphoneinfo").hide();
	}else{
		jQuery(".headercontactphoneopen").hide();
        jQuery(".headercontactphoneinfo").show();
	}
}
function pdPerformTimestampActions(jsTimestamp) {
	pdSupportIsOpenDisplay(jsTimestamp);
}
