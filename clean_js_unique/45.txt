const express = require('express');
const router = express.Router();
const path = require('path');
const User = require('../models/user-model.js');
const mongoConfig = require('../../../../config/lib/mongodb');
const jwt = require('jsonwebtoken');

router.post('/register', (req, res) => {
  var newUser = new User({
    name: {
      firstname: req.body.firstname,
      lastname: req.body.lastname,
    },
    email: req.body.email,
    username: req.body.username,
    password: req.body.password,
  });
  User.createUser(newUser, (err) => {
    if (err) {
      console.error('Something went wrong with registering');
      console.error(err);
      res.send({success: false, error: 'Check the console for error'});
    } else {
      console.log('we did it reddit, registered');
      res.send({success: true, msg: 'Created a user'});
    }
  });
});

router.post('/auth', (req, res) => {
  const username = req.body.username;
  const password = req.body.password;

  User.getUserByUsername(username, (err, user) => { // TODO Clean this up
    if (err) {
      console.error('Something went wrong with authenticating');
    } else {
      if (!user) {
        return res.json({sucess: false, error: 'User not found'});
      } else {
        User.checkPassword(password, user.password, (err, isMatch) => {
          if (err) {
            console.error('Something went wrong with checking the password');
            console.error(err);
          } else {
            if (isMatch) {
              const token = jwt.sign(user.toObject(), mongoConfig.secret, {
                expiresIn: 9999999999,
              });
              res.json({
                success: true,
                token: 'Bearer ' + token,
                user: {
                  id: user._id,
                  name: user.name,
                  username: user.username,
                },
              });
            } else {
              res.json({success: false, err: 'Incorrect password'});
            }
          }
        });
      }
    }
  });
});

module.exports = router;
